-- Procedure
CREATE OR REPLACE PROCEDURE proc_Grade(
    r NUMBER,
    n VARCHAR2,
    m NUMBER
)
IS
    c VARCHAR2(30);
BEGIN
    IF m >= 990 THEN
        c := 'Distinction';
    ELSIF m >= 900 THEN
        c := 'First Class';
    ELSIF m >= 825 THEN
        c := 'Higher Second Class';
    ELSE
        c := 'Fail';
    END IF;

    INSERT INTO Result VALUES(r,n,c);
END;
/
--then this
-- Calling Procedure
BEGIN
    FOR i IN (SELECT * FROM Stud_Marks)
    LOOP
        proc_Grade(i.Roll,i.Name,i.Total_Marks);
    END LOOP;
END;
